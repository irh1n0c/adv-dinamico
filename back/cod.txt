import React, { useState, useEffect } from "react";

const FormContent = () => {
  // Estado para manejar los valores del formulario
  const [formData, setFormData] = useState({
    quotationNumber: { number1: "", number2: "" },
    fecha: "",
    clientInfo: { razonSocial: "", ruc: "", area: "", atencion: "" },
    quotationDetails: { validezOferta: "", condicionPago: "", moneda: "soles", impuestos: "" },
    saludo: "Es grato saludarlo...",
  });

  // Actualiza el estado al cambiar los campos
  const handleInputChange = (e, section, field) => {
    const { value } = e.target;
    setFormData((prev) => {
      if (section) {
        return {
          ...prev,
          [section]: { ...prev[section], [field]: value },
        };
      }
      return { ...prev, [field]: value };
    });
  };

  // Actualiza la fecha actual automáticamente
  useEffect(() => {
    const currentDate = new Date().toLocaleDateString();
    setFormData((prev) => ({ ...prev, fecha: currentDate }));
  }, []);

  // Función para buscar cotización
  const handleSearchQuotation = () => {
    const searchNumber = formData.quotationNumber.number1 + '-' + formData.quotationNumber.number2;
    // Aquí deberías realizar la búsqueda de la cotización, por ejemplo, en una base de datos o almacenamiento
    alert(`Buscando cotización con número: ${searchNumber}`);
    // Para demostrar, pongamos datos ficticios si la cotización existe
    setFormData((prev) => ({
      ...prev,
      clientInfo: { razonSocial: "Cliente Ejemplo", ruc: "12345678901", area: "Ventas", atencion: "Juan Pérez" },
      quotationDetails: { validezOferta: "30 días", condicionPago: "Contado", moneda: "soles", impuestos: "18%" },
    }));
  };

  // Función para guardar cotización
  const handleSaveQuotation = () => {
    // Aquí puedes guardar los datos en almacenamiento local o enviarlos a una API
    localStorage.setItem('cotizacion', JSON.stringify(formData));
    alert("Cotización guardada correctamente.");
  };

  return (
    <div id="form-content">
      {/* Encabezado */}
      <div className="grid grid-cols-1 md:grid-cols-3 gap-4">
        <div className="flex items-center justify-center">
          <img src="images/logo_adv.jpg" alt="Imagen Izquierda" className="h-24 object-cover" />
        </div>

        <div className="flex flex-col items-center justify-center">
          <div className="text-center font-bold">GRUPO ADV SAC.</div>
          <div className="text-center text-xs">
            Dirección Fiscal: Mza. H Lote 5 P.J. El Triunfo Zn. A – La Joya-Arequipa
            <br />
            Almacén: Cal. Rodriguez Ballón 711-Miraflores-Arequipa
            <br />
            RUC 20605967729
          </div>
          <div className="text-center text-[10px] text-blue-500">
            Correo: ventas@grupo-adv.com o gerencia@grupo-adv.com
          </div>
        </div>

        <div className="flex items-center justify-center">
          <img src="images/iso_logo.png" alt="Imagen Derecha" className="h-24 object-cover" />
        </div>
      </div>

      {/* Formulario */}
      <div className="flex justify-end p-8">
        <div className="w-full md:w-1/3 space-y-4">
          <div className="flex items-center">
            <label className="w-1/4 text-sm font-medium">Cotización N°</label>
            <div className="w-1/2 flex gap-2">
              <input
                type="number"
                value={formData.quotationNumber.number1}
                onChange={(e) => handleInputChange(e, "quotationNumber", "number1")}
                className="w-1/2 px-3 py-2 border rounded-md"
                placeholder="000"
              />
              <input
                type="number"
                value={formData.quotationNumber.number2}
                onChange={(e) => handleInputChange(e, "quotationNumber", "number2")}
                className="w-1/2 px-3 py-2 border rounded-md"
                placeholder="000"
              />
            </div>
          </div>
          <div className="flex items-center">
            <label className="w-1/4 text-sm font-medium">Fecha Actual</label>
            <input
              type="text"
              value={formData.fecha}
              className="w-1/2 px-3 py-2 border rounded-md"
              readOnly
            />
          </div>
        </div>
      </div>

      {/* Información del cliente */}
      <div className="grid grid-cols-1 md:grid-cols-2 gap-8 p-8">
        <div className="space-y-4">
          <div className="flex items-center">
            <label className="w-1/6 text-sm font-medium">Razón Social</label>
            <input
              type="text"
              value={formData.clientInfo.razonSocial}
              onChange={(e) => handleInputChange(e, "clientInfo", "razonSocial")}
              className="w-2/3 px-3 py-2 border rounded-md"
              placeholder="Ingrese Razón Social"
            />
          </div>
          <div className="flex items-center">
            <label className="w-1/6 text-sm font-medium">RUC</label>
            <input
              type="text"
              value={formData.clientInfo.ruc}
              onChange={(e) => handleInputChange(e, "clientInfo", "ruc")}
              className="w-2/3 px-3 py-2 border rounded-md"
              placeholder="Ingrese RUC"
            />
          </div>
          <div className="flex items-center">
            <label className="w-1/6 text-sm font-medium">Área</label>
            <input
              type="text"
              value={formData.clientInfo.area}
              onChange={(e) => handleInputChange(e, "clientInfo", "area")}
              className="w-2/3 px-3 py-2 border rounded-md"
              placeholder="Ingrese Área"
            />
          </div>
          <div className="flex items-center">
            <label className="w-1/6 text-sm font-medium">Atención</label>
            <input
              type="text"
              value={formData.clientInfo.atencion}
              onChange={(e) => handleInputChange(e, "clientInfo", "atencion")}
              className="w-2/3 px-3 py-2 border rounded-md"
              placeholder="Ingrese Atención"
            />
          </div>
        </div>

        {/* Detalles de la cotización */}
        <div className="space-y-4">
          <div className="flex items-center">
            <label className="w-1/6 text-sm font-medium">Validez de Oferta</label>
            <input
              type="text"
              value={formData.quotationDetails.validezOferta}
              onChange={(e) => handleInputChange(e, "quotationDetails", "validezOferta")}
              className="w-2/3 px-3 py-2 border rounded-md"
              placeholder="Ingrese Validez de Oferta"
            />
          </div>
          <div className="flex items-center">
            <label className="w-1/6 text-sm font-medium">Condición de Pago</label>
            <input
              type="text"
              value={formData.quotationDetails.condicionPago}
              onChange={(e) => handleInputChange(e, "quotationDetails", "condicionPago")}
              className="w-2/3 px-3 py-2 border rounded-md"
              placeholder="Ingrese Condición de Pago"
            />
          </div>
          <div className="flex items-center">
            <label className="w-1/6 text-sm font-medium">Moneda</label>
            <select
              value={formData.quotationDetails.moneda}
              onChange={(e) => handleInputChange(e, "quotationDetails", "moneda")}
              className="w-2/3 px-3 py-2 border rounded-md"
            >
              <option value="soles">Soles</option>
              <option value="dolares">Dólares</option>
              <option value="euros">Euros</option>
            </select>
          </div>
          <div className="flex items-center">
            <label className="w-1/6 text-sm font-medium">Impuestos</label>
            <input
              type="text"
              value={formData.quotationDetails.impuestos}
              onChange={(e) => handleInputChange(e, "quotationDetails", "impuestos")}
              className="w-2/3 px-3 py-2 border rounded-md"
              placeholder="Ingrese Impuestos"
            />
          </div>
        </div>
      </div>

      {/* Saludo */}
      <div className="px-8">
        <input
          type="text"
          value={formData.saludo}
          onChange={(e) => handleInputChange(e, null, "saludo")}
          className="w-full px-3 py-2 border rounded-md"
        />
      </div>

      {/* Botones */}
      <div className="flex justify-end gap-4 p-8">
        <button onClick={handleSearchQuotation} className="px-4 py-2 bg-blue-500 text-white rounded-md">
          Buscar Cotización
        </button>
        <button onClick={handleSaveQuotation} className="px-4 py-2 bg-green-500 text-white rounded-md">
          Guardar Cotización
        </button>
      </div>
    </div>
  );
};

export default FormContent;
